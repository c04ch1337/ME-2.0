# ============================================================================
# ME-2.0 Bolna Telephony Platform - Environment Configuration
# ============================================================================
# This configuration file contains all environment variables needed for the
# complete 5-service Bolna telephony stack:
# - bolna-app (main Bolna application)
# - redis (data store)
# - ngrok (tunnel for webhooks)
# - twilio-app (Twilio telephony integration)
# - plivo-app (Plivo telephony integration - alternative to Twilio)
#
# For detailed setup instructions, see: docs/Manual Step-by-Step Installation.md
# ============================================================================

# ============================================================================
# TELEPHONY PROVIDER CONFIGURATION
# ============================================================================

# Twilio Configuration (Primary telephony provider)
# Required for: Voice calls via Twilio
TWILIO_ACCOUNT_SID=your_sid
TWILIO_AUTH_TOKEN=your_token
TWILIO_PHONE_NUMBER=+1234567890

# Plivo Configuration (Alternative telephony provider)
# Required for: Voice calls via Plivo (if not using Twilio)
# Note: Either Twilio OR Plivo is required for telephony functionality
PLIVO_AUTH_ID=your_plivo_auth_id
PLIVO_AUTH_TOKEN=your_plivo_auth_token
PLIVO_PHONE_NUMBER=+1234567890

# ============================================================================
# BOLNA APPLICATION CONFIGURATION
# ============================================================================

# Bolna Service Ports
# These ports are used by different Bolna service components
BOLNA_APP_PORT=5001
BOLNA_TWILIO_PORT=8001
BOLNA_PLIVO_PORT=8002

# Bolna Environment
# Options: development, production
BOLNA_ENV=production

# ============================================================================
# NGROK CONFIGURATION
# ============================================================================
# Ngrok creates secure tunnels to expose local services for webhook callbacks
# Required for: Receiving telephony provider webhooks (Twilio/Plivo)
# Get your authtoken from: https://dashboard.ngrok.com/get-started/your-authtoken

NGROK_AUTHTOKEN=your_ngrok_auth_token_here
# Ngrok region options: us, eu, ap, au, sa, jp, in
NGROK_REGION=us

# ============================================================================
# REDIS CONFIGURATION
# ============================================================================
# Redis is used for caching and session management in Bolna
# Required for: All Bolna operations

REDIS_HOST=redis
REDIS_PORT=6379
# Leave empty if Redis has no password (default for local development)
REDIS_PASSWORD=
REDIS_DB=0

# ============================================================================
# AI MODEL CONFIGURATION
# ============================================================================

# Ollama Model (for local LLM inference)
# Optional: Only needed if using Ollama for local models
OLLAMA_MODEL=llama3

# Mem0 Configuration (AI memory layer)
# Required for: Persistent conversation context
MEM0_USER_ID=me

# DreamLit API (if using DreamLit services)
# Optional: Only needed if integrating DreamLit
DREAMLIT_API_KEY=your_key

# Google OAuth (if using Google authentication)
# Optional: Only needed for Google sign-in features
GOOGLE_CLIENT_ID=your_id

# ============================================================================
# GPU CONFIGURATION
# ============================================================================
# Specify which GPU devices to use for model inference
# Optional: Only needed if running models on GPU

CUDA_VISIBLE_DEVICES=2,3

# ============================================================================
# APPLICATION SETTINGS
# ============================================================================

# Logging level options: DEBUG, INFO, WARNING, ERROR, CRITICAL
LOG_LEVEL=INFO

# Storage paths for models and data
MODEL_PATH=/app/models
DATA_PATH=/app/data

# Application environment
# Options: development, staging, production
APP_ENV=development

# ============================================================================
# OPENAI CONFIGURATION
# ============================================================================
# OpenAI is required for mem0 by default and used by Bolna via OpenAI/LiteLLM
# Required for: Core AI functionality, embeddings, and conversation

OPENAI_API_KEY=your_openai_api_key

# Optional: Override base URL for OpenAI-compatible providers
# Use this for: Azure OpenAI, OpenRouter, or self-hosted OpenAI proxies
OPENAI_BASE_URL=

# To avoid using OpenAI for embeddings in mem0, use a local provider
# Uncomment the line below to switch to FastEmbed (local embeddings)
# MEM0_EMBEDDINGS_PROVIDER=fastembed

# ============================================================================
# AWS CONFIGURATION
# ============================================================================
# AWS credentials for S3 access and other AWS services
# Optional: Only needed if using AWS services for storage or other features

AWS_ACCESS_KEY_ID=your_aws_access_key
AWS_SECRET_ACCESS_KEY=your_aws_secret_key
AWS_DEFAULT_REGION=us-east-1

# ============================================================================
# VARIABLE REQUIREMENTS SUMMARY
# ============================================================================
# 
# REQUIRED for basic Bolna setup (OpenAI-only):
#   - OPENAI_API_KEY
#   - REDIS_HOST, REDIS_PORT
#   - BOLNA_APP_PORT
#
# REQUIRED for full telephony functionality:
#   - OPENAI_API_KEY
#   - REDIS_HOST, REDIS_PORT
#   - BOLNA_APP_PORT, BOLNA_TWILIO_PORT (or BOLNA_PLIVO_PORT)
#   - Either TWILIO_* OR PLIVO_* variables (choose one provider)
#   - NGROK_AUTHTOKEN (for webhook callbacks)
#
# OPTIONAL:
#   - AWS_* (only if using S3 or other AWS services)
#   - OLLAMA_MODEL (only if using local Ollama models)
#   - DREAMLIT_API_KEY (only if using DreamLit)
#   - GOOGLE_CLIENT_ID (only if using Google OAuth)
#   - CUDA_VISIBLE_DEVICES (only if using GPU acceleration)
#   - MEM0_EMBEDDINGS_PROVIDER (to use local embeddings instead of OpenAI)
#
# For detailed configuration and setup instructions, see:
#   docs/Manual Step-by-Step Installation.md
# ============================================================================
